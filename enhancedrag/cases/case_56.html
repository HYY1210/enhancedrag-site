Raw Evidence and Source Files¶

runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'wca.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.001
 8app = gala.Application(all_info, dt)
 9neighbor_list = gala.NeighborList(all_info, 2 ** (1.0 / 6.0), 0.1)
10lj = gala.LJForce(all_info, neighbor_list, 2 ** (1.0 / 6.0))
11lj.setParams('A', 'A', 1.0, 1.0, 1.0)
12lj.setEnergy_shift()
13app.add(lj)
14group = gala.ParticleSet(all_info, 'all')
15comp_info = gala.ComputeInfo(all_info, group)
16nh = gala.NoseHooverNVT(all_info, group, comp_info, 1.0, 0.5)
17app.add(nh)
18sort_method = gala.Sort(all_info)
19sort_method.setPeriod(1000)
20app.add(sort_method)
21xml = gala.XMLDump(all_info, 'particles')
22xml.setPeriod(10000)
23app.add(xml)
24DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
25DInfo.setPeriod(300)
26app.add(DInfo)
27app.run(20000)
28neighbor_list.printStats()
