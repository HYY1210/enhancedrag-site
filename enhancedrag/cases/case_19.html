Raw Evidence and Source Files¶

runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'patchyinit.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.002
 8app = gala.Application(all_info, dt)
 9neighbor_list = gala.NeighborList(all_info, 1.0, 0.1)
10neighbor_list.setBlockSize(96)
11ani = gala.AniForce(all_info, neighbor_list, 1.0)
12ani.setParams('A', 'A', 396.0, 2.0)
13ani.setParams('A', 'B', 396.0, 2.0)
14ani.setParams('B', 'B', 396.0, 2.0)
15ani.setPatches('patch-3.log')
16ani.setBlockSize(96)
17app.add(ani)
18group = gala.ParticleSet(all_info, 'all')
19comp_info = gala.ComputeInfo(all_info, group)
20bgroup = gala.ParticleSet(all_info, 'body')
21rigidnvt = gala.NVTRigid(all_info, bgroup, 1.0, 0.2)
22app.add(rigidnvt)
23nbgroup = gala.ParticleSet(all_info, 'non_body')
24comp_info_nb = gala.ComputeInfo(all_info, nbgroup)
25nh = gala.NoseHooverNVT(all_info, nbgroup, comp_info_nb, 1.0, 1.0)
26app.add(nh)
27sort_method = gala.Sort(all_info)
28sort_method.setPeriod(300)
29app.add(sort_method)
30DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
31DInfo.setPeriod(10000)
32DInfo.dumpAnisotropy()
33app.add(DInfo)
34ZeroMomentum = gala.ZeroMomentum(all_info)
35ZeroMomentum.setPeriod(10000)
36app.add(ZeroMomentum)
37xml = gala.XMLDump(all_info, 'particles')
38xml.setPeriod(500000)
39xml.setOutputQuaternion(True)
40xml.setOutputPatch(ani)
41app.add(xml)
42app.run(8000000)
43neighbor_list.printStats()
