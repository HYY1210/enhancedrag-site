runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'double-layer.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.005
 8app = gala.Application(all_info, dt)
 9scf = gala.MDSCFForce(all_info, 42, 42, 42, 0.05)
10scf.setParams('P', 'P', 0)
11scf.setParams('N', 'N', 0)
12scf.setParams('W', 'W', 0)
13scf.setParams('P', 'N', 0)
14scf.setParams('P', 'W', 0)
15scf.setParams('N', 'W', 0)
16scf.setPeriodScf(1, 1)
17app.add(scf)
18pfme = gala.PFMEForce(all_info, 42, 42, 42, 4.2, 20)
19pfme.setPeriodPFME(1, 1)
20app.add(pfme)
21group = gala.ParticleSet(all_info, 896, 20504)
22comp_info = gala.ComputeInfo(all_info, group)
23Temperature = 300.0
24T = Temperature * 8.3143 / 1000.0
25thermo = gala.AndersenNVT(all_info, group, T, 7.0, 1234)
26app.add(thermo)
27sort_method = gala.Sort(all_info)
28sort_method.setPeriod(400)
29app.add(sort_method)
30ZeroMomentum = gala.ZeroMomentum(all_info)
31ZeroMomentum.setPeriod(0)
32app.add(ZeroMomentum)
33DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
34DInfo.setPeriod(200)
35DInfo.dumpVirialEnergy(pfme)
36app.add(DInfo)
37mol2 = gala.MOL2Dump(all_info, 'particles')
38mol2.setPeriod(0)
39app.add(mol2)
40xml = gala.XMLDump(all_info, 'particles')
41xml.setPeriod(100000)
42xml.setOutputVelocity(True)
43xml.setOutputMass(True)
44xml.setOutputImage(True)
45xml.setOutputBond(True)
46app.add(xml)
47app.run(100000000)
