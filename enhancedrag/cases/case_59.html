 11def rotMatrix(a, b, c):
 12    bcx = c[0] - b[0]
 13    bcy = c[1] - b[1]
 14    bcz = c[2] - b[2]
 15    bax = b[0] - a[0]
 16    bay = b[1] - a[1]
 17    baz = b[2] - a[2]

 20    x2 = bcz

 22    z1 = bcz * bax - baz * bcx

 24    y0 = z1 * x2 - x1 * z2
 25    y1 = z2 * x0 - x2 * z0
 26    y2 = z0 * x1 - x0 * z1
 27    lx = math.sqrt(x0 * x0 + x1 * x1 + x2 * x2)
