 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'DOPE.xml'
 4build_method = gala.XMLReader(filename)

14scf.setParams('Qd', 'C3', 11.7005959335)
15scf.setParams('Qd', 'P4', -3.59944543508)

17scf.setParams('Na', 'Qa', 3.0002168673)
18scf.setParams('Na', 'C1', 6.29855584806)
19scf.setParams('Na', 'C3', 6.29855584806)
20scf.setParams('Na', 'P4', 3.0002168673)
21scf.setParams('Qa', 'Qa', 0.0)
22scf.setParams('Qa', 'C1', 13.4996423272)
