runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'water.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.001
 8app = gala.Application(all_info, dt)
 9neighbor_list = gala.NeighborList(all_info, 0.9, 0.09)
10neighbor_list.exclusion(['constraint', 'vsite'])
11lj = gala.LJEwaldForce(all_info, neighbor_list, 0.9)
12lj.setParams('OW', 'OW', 0.64852, 0.315365, 1.0)
13lj.setParams('HW', 'HW', 0.0, 0.47, 1.0)
14lj.setParams('MW', 'MW', 0.0, 0.47, 1.0)
15lj.setParams('OW', 'HW', 0.0, 0.47, 1.0)
16lj.setParams('OW', 'MW', 0.0, 0.47, 1.0)
17lj.setParams('HW', 'MW', 0.0, 0.47, 1.0)
18lj.setEnergy_shift()
19lj.setDispVirialCorr(True)
20app.add(lj)
21bondconstraint = gala.BondConstraint(all_info)
22bondconstraint.setParams('oh', 0.09572)
23bondconstraint.setParams('hh', 0.15139)
24bondconstraint.setExpansionOrder(4)
25bondconstraint.setNumIters(2)
26app.add(bondconstraint)
27groupC = gala.ParticleSet(all_info, 'charge')
28pppm = gala.PPPMForce(all_info, neighbor_list, groupC)
29pppm.setParams(0.12, 4, 0.9)
30app.add(pppm)
31vs = gala.Vsite(all_info)
32vs.setParams('v', 0.128012065, 0.128012065, 0.0, gala.VST.v3)
33app.add(vs)
34Temperature = 298.0
35T = Temperature * 8.3143 / 1000.0
36group = gala.ParticleSet(all_info, 'all')
37comp_info = gala.ComputeInfo(all_info, group)
38npt = gala.NPTMTK(all_info, group, comp_info, comp_info, T, 0.0602, 1.0, 1.0)
39app.add(npt)
40sort_method = gala.Sort(all_info)
41sort_method.setPeriod(800)
42app.add(sort_method)
43ZeroMomentum = gala.ZeroMomentum(all_info)
44ZeroMomentum.setPeriod(100000)
45app.add(ZeroMomentum)
46DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
47DInfo.setPeriod(1000)
48DInfo.dumpVirialEnergy(lj)
49DInfo.dumpVirialEnergy(pppm)
50app.add(DInfo)
51xml = gala.XMLDump(all_info, 'p')
52xml.setPeriod(10000)
53xml.setOutput(['constraint', 'vsite', 'mass', 'velocity', 'charge', 'image'])
54app.add(xml)
55app.run(1000000)
56neighbor_list.printStats()
