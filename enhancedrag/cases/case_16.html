Raw Evidence and Source Files¶

runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'gbinit.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.0002
 8app = gala.Application(all_info, dt)
 9neighbor_list = gala.NeighborList(all_info, 4.0, 0.8)
10neighbor_list.addExclusionsFromBodies()
11pbgb = gala.PBGBForce(all_info, neighbor_list)
12pbgb.setGUM(1.0, 3.0, 1.0)
13pbgb.setParams('B', 'B', 2.0, 1.0, 4.0)
14pbgb.setParams('A', 'A', 2.0, 1.0, 2.5)
15pbgb.setParams('A', 'B', 1.0, 1.0, 3.25)
16pbgb.setAspheres('patch-2.log')
17pbgb.setPatches('patch-2.log')
18app.add(pbgb)
19group = gala.ParticleSet(all_info, 'all')
20comp_info = gala.ComputeInfo(all_info, group)
21bgroup = gala.ParticleSet(all_info, 'body')
22comp_info_b = gala.ComputeInfo(all_info, bgroup)
23rigidnpt = gala.NPTRigid(all_info, bgroup, comp_info_b, comp_info, 2.4, 8.0, 0.5, 0.5)
24app.add(rigidnpt)
25nbgroup = gala.ParticleSet(all_info, 'non_body')
26comp_info_nb = gala.ComputeInfo(all_info, nbgroup)
27npt = gala.NPT(all_info, nbgroup, comp_info_nb, comp_info, 2.4, 8.0, 0.5, 0.5)
28app.add(npt)
29sort_method = gala.Sort(all_info)
30sort_method.setPeriod(300)
31app.add(sort_method)
32DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
33DInfo.setPeriod(500)
34DInfo.dumpAnisotropy()
35app.add(DInfo)
36ZeroMomentum = gala.ZeroMomentum(all_info)
37ZeroMomentum.setPeriod(50000)
38app.add(ZeroMomentum)
39xml = gala.XMLDump(all_info, 'particles')
40xml.setPeriod(20000)
41xml.setOutputQuaternion(True)
42xml.setOutputBody(True)
43xml.setOutputMass(True)
44xml.setOutputImage(True)
45xml.setOutputEllipsoid(pbgb)
46app.add(xml)
47app.run(10000000)
48neighbor_list.printStats()
