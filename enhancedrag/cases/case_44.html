 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'SDS.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.03
 8app = gala.Application(all_info, dt)
 9scf = gala.MDSCFForce(all_info, 50, 50, 50, 0.1)
10scf.setParams('S', 'S', 0.0)
11scf.setParams('S', 'C', 13.5)
12scf.setParams('S', 'Na', 0.0)
13scf.setParams('S', 'W', -3.6)
14scf.setParams('C', 'C', 0.0)
15scf.setParams('C', 'Na', 13.5)
16scf.setParams('C', 'W', 33.75)
17scf.setParams('Na', 'Na', 0.0)
18scf.setParams('Na', 'W', 0.0)
19scf.setParams('W', 'W', 0.0)
20scf.setPeriodScf(1, 100)
21app.add(scf)
22pfme = gala.PFMEForce(all_info, 50, 50, 50, 2.35, 80.0)
23pfme.setPeriodPFME(1, 1)
24app.add(pfme)
25bondforce = gala.BondForceHarmonic(all_info)
26bondforce.setParams('S-C', 10000.0, 0.404)
27bondforce.setParams('C-C', 1250.0, 0.47)
28app.add(bondforce)
29angleforce = gala.AngleForceHarmonicCos(all_info)
30angleforce.setParams('S-C-C', 15.0, 170.0)
31angleforce.setParams('C-C-C', 25.0, 180.0)
32app.add(angleforce)
33group = gala.ParticleSet(all_info, 'all')
34comp_info = gala.ComputeInfo(all_info, group)
35Temperature = 298.0

37thermo = gala.AndersenNVT(all_info, group, T, 7.0, 1234)
38app.add(thermo)
39sort_method = gala.Sort(all_info)
40sort_method.setPeriod(400)
41app.add(sort_method)
42DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
43DInfo.setPeriod(400)
44app.add(DInfo)
45mol2 = gala.MOL2Dump(all_info, 'particles')
46mol2.setPeriod(0)
47app.add(mol2)
48xml = gala.XMLDump(all_info, 'particles')
49xml.setPeriod(100000)
