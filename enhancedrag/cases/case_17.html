Raw Evidence and Source Files¶

runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'ellip.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.001
 8app = gala.Application(all_info, dt)
 9neighbor_list = gala.NeighborList(all_info, 5.0, 0.8)
10pbgb = gala.PBGBForce(all_info, neighbor_list)
11pbgb.setGUM(1.0, 3.0, 1.0)
12pbgb.setParams('A', 'A', 1.0, 1.0, 5.0)
13pbgb.setParams('A', 'B', 1.0, 1.0, 5.0)
14pbgb.setParams('B', 'B', 1.0, 1.0, 5.0)
15pbgb.setAspheres('patch-2.log')
16bf = gala.BondForceHarmonicEllipsoid(all_info)
17bf.setSpots('patch-2.log')
18bf.setParams('b0', 1000.0, 3.0)
19app.add(bf)
20af = gala.AngleForceHarmonicEllipsoid(all_info)
21af.setSpots('patch-2.log')
22af.setParams('a0', 500.0, 60.0)
23af.setParams('a1', 500.0, 150.0)
24app.add(af)
25df = gala.DihedralForceHarmonicEllipsoid(all_info)
26df.setSpots('patch-2.log')
27df.setParams('d0', 100.0, 0.0)
28app.add(df)
29group = gala.ParticleSet(all_info, 'all')
30comp_info = gala.ComputeInfo(all_info, group)
31bgroup = gala.ParticleSet(all_info, 'body')
32rigidnvt = gala.NVERigid(all_info, bgroup)
33app.add(rigidnvt)
34nbgroup = gala.ParticleSet(all_info, 'non_body')
35Nh = gala.NVE(all_info, nbgroup)
36app.add(Nh)
37sort_method = gala.Sort(all_info)
38sort_method.setPeriod(300)
39app.add(sort_method)
40DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
41DInfo.setPeriod(500)
42DInfo.dumpAnisotropy()
43app.add(DInfo)
44ZeroMomentum = gala.ZeroMomentum(all_info)
45ZeroMomentum.setPeriod(10000)
46app.add(ZeroMomentum)
47xml = gala.XMLDump(all_info, 'particles')
48xml.setPeriod(1000)
49xml.setOutputQuaternion(True)
50xml.setOutputEllipsoid(pbgb)
51xml.setOutputEllipsoid(bf)
52xml.setOutputBond(True)
53app.add(xml)
54app.run(100000)
55neighbor_list.printStats()
