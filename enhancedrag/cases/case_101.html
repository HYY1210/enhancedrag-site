runDynamics.py¶

 1import math
 2from poetry import cu_gala as gala
 3from poetry import _options
 4filename = 'init_one.xml'
 5build_method = gala.XMLReader(filename)
 6perform_config = gala.PerformConfig(_options.gpu)
 7all_info = gala.AllInfo(build_method, perform_config)
 8dt = 0.001
 9app = gala.Application(all_info, dt)
10neighbor_list = gala.NeighborList(all_info, 3.0, 0.2)
11gem = gala.GEMForce(all_info, neighbor_list, 3.0)
12gem.setParams('A', 'A', 1.0, 1.0, 4.0)
13app.add(gem)
14group = gala.ParticleSet(all_info, 'all')
15comp_info = gala.ComputeInfo(all_info, group)
16nh = gala.NoseHooverNVT(all_info, group, comp_info, 0.02, 0.5)
17app.add(nh)
18ZeroMomentum = gala.ZeroMomentum(all_info)
19ZeroMomentum.setPeriod(20000)
20app.add(ZeroMomentum)
21sort_method = gala.Sort(all_info)
22sort_method.setPeriod(1000)
23app.add(sort_method)
24DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
25DInfo.setPeriod(1000)
26app.add(DInfo)
27xml = gala.XMLDump(all_info, 'particle')
28xml.setPeriod(10000000)
29app.add(xml)
30v = gala.VariantLinear()
31v.setPoint(0, 0.07)
32v.setPoint(20000000, 0.07)
33v.setPoint(30000000, 0.02)
34v.setPoint(40000000, 0.02)
35v.setPoint(50000000, 0.09)

37v.setPoint(70000000, 0.02)

41v.setPoint(110000000, 0.02)
42nh.setT(v)
