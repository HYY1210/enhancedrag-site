runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'ab.xml'
 4randomnum = 12340
 5build_method = gala.XMLReader(filename)
 6perform_config = gala.PerformConfig(_options.gpu)
 7all_info = gala.AllInfo(build_method, perform_config)
 8dt = 0.04
 9app = gala.Application(all_info, dt)
10neighbor_list = gala.NeighborList(all_info, 1.0, 0.05)
11dpd = gala.DPDForce(all_info, neighbor_list, 1.0, randomnum)
12dpd.setParams('A', 'A', 25.0, 3.0)
13dpd.setParams('B', 'B', 25.0, 3.0)
14dpd.setParams('A', 'B', 70.0, 3.0)
15app.add(dpd)
16group = gala.ParticleSet(all_info, 'all')
17comp_info = gala.ComputeInfo(all_info, group)
18Gwvv = gala.DPDGWVV(all_info, group)
19app.add(Gwvv)
20sort_method = gala.Sort(all_info)
21sort_method.setPeriod(80)
22app.add(sort_method)
23DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
24DInfo.setPeriod(200)
25app.add(DInfo)
26dcd = gala.DCDDump(all_info, 'particles', True)
27dcd.setPeriod(50000)
28dcd.unwrap(True)
29app.add(dcd)
30xml = gala.XMLDump(all_info, 'particles')
31xml.setPeriod(100000)
32app.add(xml)
33app.run(1000000)
34neighbor_list.printStats()
