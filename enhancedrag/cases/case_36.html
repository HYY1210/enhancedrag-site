runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'AB-100.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.005
 8app = gala.Application(all_info, dt)
 9scf = gala.MDSCFForce(all_info, 42, 42, 42, 0.05)
10scf.setParams('A', 'A', 0)
11scf.setParams('B', 'B', 0)
12scf.setParams('C', 'C', 0)
13scf.setParams('W', 'W', 0)
14scf.setParams('A', 'B', 0)
15scf.setParams('A', 'C', 0)
16scf.setParams('A', 'W', 0)
17scf.setParams('B', 'C', 0)
18scf.setParams('B', 'W', 0)
19scf.setParams('C', 'W', 0)
20scf.setPeriodScf(1, 1)
21app.add(scf)
22pfme = gala.PFMEForce(all_info, 42, 42, 42, 4.2, 11.7)
23pfme.setPeriodPFME(1, 1)
24app.add(pfme)
25bondforce = gala.BondForceHarmonic(all_info)
26bondforce.setParams('A-B', 1000, 0.4)
27bondforce.setParams('A-A', 1000, 0.4)
28app.add(bondforce)
29group = gala.ParticleSet(all_info, 'all')
30comp_info = gala.ComputeInfo(all_info, group)
31Temperature = 300.0
32T = Temperature * 8.3143 / 1000.0
33thermo = gala.AndersenNVT(all_info, group, T, 7.0, 1234)
34app.add(thermo)
35sort_method = gala.Sort(all_info)
36sort_method.setPeriod(400)
37app.add(sort_method)
38ZeroMomentum = gala.ZeroMomentum(all_info)
39ZeroMomentum.setPeriod(0)
40app.add(ZeroMomentum)
41DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
42DInfo.setPeriod(200)
43DInfo.dumpVirialEnergy(pfme)
44app.add(DInfo)
45mol2 = gala.MOL2Dump(all_info, 'particles')
46mol2.setPeriod(0)
47app.add(mol2)
48xml = gala.XMLDump(all_info, 'particles')
49xml.setPeriod(100000)
50xml.setOutputVelocity(True)
51xml.setOutputMass(True)
52xml.setOutputImage(True)
53xml.setOutputBond(True)
54app.add(xml)
55app.run(10000000)
