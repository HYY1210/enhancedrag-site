runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'pmmapeo.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.03
 8app = gala.Application(all_info, dt)
 9scf = gala.MDSCFForce(all_info, 24 * 2, 24 * 2, 24 * 2, 0.05)
10scf.setParams('A', 'A', 0.0)
11scf.setParams('A', 'B', 0.0)
12scf.setParams('A', 'C', 2.0)
13scf.setParams('A', 'D', 0.0)
14scf.setParams('B', 'B', 0.0)
15scf.setParams('B', 'C', 2.0)
16scf.setParams('B', 'D', 0.0)
17scf.setParams('C', 'C', 0.0)
18scf.setParams('C', 'D', 0.0)
19scf.setParams('D', 'D', 0.0)
20scf.setPeriodScf(1, 10)
21app.add(scf)
22pfme = gala.PFMEForce(all_info, 24 * 2, 24 * 2, 24 * 2, 4.2, 7.3)
23pfme.setPeriodPFME(1, 10)
24app.add(pfme)
25bondforce = gala.BondForceHarmonic(all_info)

29bondforce.setParams('C-C', 17000.0, 0.33)
30app.add(bondforce)
31angleforce = gala.AngleForceHarmonic(all_info)
32angleforce.setParams('A-A-A', 33.0, 120.0)
33angleforce.setParams('A-A-B', 36.0, 95.0)
34angleforce.setParams('B-A-C', 36.0, 95.0)

36angleforce.setParams('C-C-C', 85.0, 130.0)
37app.add(angleforce)

39angleforce1.setParams('A-A-A', 82.0, 154.0)
40angleforce1.setParams('A-A-B', 10.3, 110.0)
41angleforce1.setParams('B-A-C', 10.3, 110.0)
42app.add(angleforce1)

46app.add(dihedralforce)

48comp_info = gala.ComputeInfo(all_info, group)

54sort_method.setPeriod(400)
55app.add(sort_method)
56ZeroMomentum = gala.ZeroMomentum(all_info)
