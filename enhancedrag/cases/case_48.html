runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'pspeo.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.03
 8app = gala.Application(all_info, dt)
 9scf = gala.MDSCFForce(all_info, 48, 48, 48, 0.05)
10scf.setParams('A', 'A', 0.0)
11scf.setParams('A', 'B', 0.0)
12scf.setParams('A', 'C', 5.0)
13scf.setParams('A', 'D', 0.0)
14scf.setParams('B', 'B', 0.0)
15scf.setParams('B', 'C', 5.0)
16scf.setParams('B', 'D', 0.0)
17scf.setParams('C', 'C', 0.0)
18scf.setParams('C', 'D', 0.0)
19scf.setParams('D', 'D', 0.0)
20scf.setPeriodScf(1, 10)
21app.add(scf)
22pfme = gala.PFMEForce(all_info, 48, 48, 48, 4.2, 7.3)
23pfme.setPeriodPFME(1, 10)
24app.add(pfme)
25bondforce = gala.BondForceHarmonic(all_info)
26bondforce.setParams('A-B', 5000.0, 0.27)
27bondforce.setParams('B-B', 5000.0, 0.27)
28bondforce.setParams('A-C', 5000.0, 0.27)
29bondforce.setParams('C-C', 5000.0, 0.33)
30app.add(bondforce)
31angleforce = gala.AngleForceHarmonic(all_info)
32angleforce.setParams('A-B-B', 100.0, 136.0)
33angleforce.setParams('B-A-B', 25.0, 120.0)
34angleforce.setParams('A-B-A', 550.0, 52.0)
35angleforce.setParams('C-C-C', 50.0, 130.0)
36app.add(angleforce)
37dihedralforce = gala.DihedralForceHarmonic(all_info)
38dihedralforce.setParams('C-C-C-C', 3.0, 180.0)
39app.add(dihedralforce)
40group = gala.ParticleSet(all_info, 'all')
41comp_info = gala.ComputeInfo(all_info, group)
42Temperature = 318.0
43T = Temperature * 8.3143 / 1000.0
44thermo = gala.AndersenNVT(all_info, group, T, 7.0, 1234)
45app.add(thermo)
46sort_method = gala.Sort(all_info)
47sort_method.setPeriod(400)
48app.add(sort_method)
49ZeroMomentum = gala.ZeroMomentum(all_info)
50ZeroMomentum.setPeriod(0)
51app.add(ZeroMomentum)
52DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
53DInfo.setPeriod(200)
54DInfo.dumpVirialEnergy(pfme)
55app.add(DInfo)
56mol2 = gala.MOL2Dump(all_info, 'particles')
57mol2.setPeriod(0)
58app.add(mol2)
59xml = gala.XMLDump(all_info, 'particles')
60xml.setPeriod(100000)
61xml.setOutputVelocity(True)
62xml.setOutputMass(True)
63xml.setOutputImage(True)
64xml.setOutputBond(True)
65app.add(xml)
66comp_info.setNdof(app.getNdof())
67app.run(20000000)
