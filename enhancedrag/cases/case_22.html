runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'S100.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.001
 8app = gala.Application(all_info, dt)
 9neighbor_list = gala.NeighborList(all_info, 2.5, 0.25)
10neighbor_list.addExclusionsFromBodies()
11lj = gala.LJForce(all_info, neighbor_list, 2.5)
12lj.setParams('A', 'A', 1.0, 1.0, 1.0, 2 ** (1.0 / 6.0))
13lj.setParams('A', 'B', 1.0, 1.0, 1.0, 2.5)
14lj.setParams('B', 'B', 1.0, 1.0, 1.0, 2 ** (1.0 / 6.0))
15lj.setEnergy_shift()
16app.add(lj)
17BondForceFENE = gala.BondForceFENE(all_info)
18BondForceFENE.setParams('B-B', 30, 1.5)
19app.add(BondForceFENE)
20group = gala.ParticleSet(all_info, 'all')
21comp_info = gala.ComputeInfo(all_info, group)
22bgroup = gala.ParticleSet(all_info, 'body')
23rigidnvt = gala.NVTRigid(all_info, bgroup, 1.0, 100)
24app.add(rigidnvt)
25nbgroup = gala.ParticleSet(all_info, 'non_body')
26comp_info_nb = gala.ComputeInfo(all_info, nbgroup)
27Nh = gala.NoseHooverNVT(all_info, nbgroup, comp_info_nb, 1.0, 0.5)
28app.add(Nh)
29DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
30DInfo.setPeriod(500)
31app.add(DInfo)
32ZeroMomentum = gala.ZeroMomentum(all_info)
33ZeroMomentum.setPeriod(50000)
34app.add(ZeroMomentum)
35sort_method = gala.Sort(all_info)
36sort_method.setPeriod(500)
37app.add(sort_method)
38xml = gala.XMLDump(all_info, 'polymer')
39xml.setPeriod(100000)
40xml.setOutputPosition(True)
41xml.setOutputImage(True)
42xml.setOutputType(True)
43xml.setOutputBond(True)
44xml.setOutputBody(True)
45app.add(xml)
46comp_info.setNdof(app.getNdof())
47app.run(200000)
48neighbor_list.printStats()
