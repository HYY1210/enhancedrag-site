Expert Note¶

Expert: 朱有亮

这是一个两嵌段线性聚合物的耗散粒子动力学模拟，分为A和B两个嵌段，其中A嵌段比例为14%，通过A嵌段比例为10%和20%的两种链长为10的分子链模型共混得到，在耗散粒子动力学模拟下，这种嵌段比例的聚合物会自发形成球状相结构，模拟嵌段共聚物的微相分离行为。

Raw Evidence and Source Files¶

runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'A1B9-A2B8.xml'
 4randomnum = 12340
 5build_method = gala.XMLReader(filename)
 6perform_config = gala.PerformConfig(_options.gpu)
 7all_info = gala.AllInfo(build_method, perform_config)
 8dt = 0.04
 9app = gala.Application(all_info, dt)
10neighbor_list = gala.NeighborList(all_info, 1.0, 0.05)
11dpd = gala.DPDForce(all_info, neighbor_list, 1.0, randomnum)
12dpd.setParams('A', 'A', 25.0, 3.0)
13dpd.setParams('A', 'B', 57.0, 3.0)
14dpd.setParams('B', 'B', 25.0, 3.0)
15app.add(dpd)
16bondforce = gala.BondForceHarmonic(all_info)
17bondforce.setParams('A-A', 4.0, 0.0)
18bondforce.setParams('A-B', 4.0, 0.0)
19bondforce.setParams('B-B', 4.0, 0.0)
20app.add(bondforce)
21group = gala.ParticleSet(all_info, 'all')
22comp_info = gala.ComputeInfo(all_info, group)
23Gwvv = gala.DPDGWVV(all_info, group)
24app.add(Gwvv)
25sort_method = gala.Sort(all_info)
26sort_method.setPeriod(80)
27app.add(sort_method)
28DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
29DInfo.setPeriod(200)
30app.add(DInfo)
31xml = gala.XMLDump(all_info, 'particles')
32xml.setPeriod(100000)
33xml.setOutputBond(True)
34app.add(xml)
35app.run(2000000)
36neighbor_list.printStats()
