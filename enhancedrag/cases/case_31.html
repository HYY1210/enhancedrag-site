runDynamics.py¶

 1from poetry import cu_gala as gala
 2from poetry import _options
 3filename = 'test.xml'
 4build_method = gala.XMLReader(filename)
 5perform_config = gala.PerformConfig(_options.gpu)
 6all_info = gala.AllInfo(build_method, perform_config)
 7dt = 0.01
 8app = gala.Application(all_info, dt)
 9scf = gala.MDSCFForce(all_info, 17, 17, 17, 0.1)
10scf.setParams('W', 'W', 0)
11scf.setParams('Na', 'Na', 0)
12scf.setParams('Cl', 'Cl', 0)
13scf.setParams('W', 'Na', 0)
14scf.setParams('W', 'Cl', 0)
15scf.setParams('Na', 'Cl', 0)
16scf.setPeriodScf(1, 1)
17scf.setBlockSizeScf(64, 64, 64, 64)
18scf.setNewVersion(True)
19app.add(scf)
20pfme = gala.PFMEForce(all_info, 17, 17, 17, 5.0, 1.0)
21pfme.setPeriodPFME(1, 1)
22pfme.setBlockSizePFME(64, 64, 64, 64)
23app.add(pfme)
24group = gala.ParticleSet(all_info, 'all')
25comp_info = gala.ComputeInfo(all_info, group)
26Temperature = 300.0
27T = Temperature * 8.3143 / 1000.0
28thermo = gala.AndersenNVT(all_info, group, T, 7.0, 1234)
29app.add(thermo)
30sort_method = gala.Sort(all_info)
31sort_method.setPeriod(400)
32app.add(sort_method)
33ZeroMomentum = gala.ZeroMomentum(all_info)
34ZeroMomentum.setPeriod(0)
35app.add(ZeroMomentum)
36DInfo = gala.DumpInfo(all_info, comp_info, 'data.log')
37DInfo.setPeriod(200)
38app.add(DInfo)
39xml = gala.XMLDump(all_info, 'particles')
40xml.setPeriod(100000)
41xml.setOutputVelocity(True)
42xml.setOutputMass(True)
43xml.setOutputImage(True)
44app.add(xml)
45comp_info.setNdof(app.getNdof())
46app.run(4000000)
